  geom_col(alpha=.5,
           show.legend=F) +
  scale_x_continuous(#limits=c(0,45e3),
                     #breaks=seq(0,45e3,5e3),
                     labels=label_number(scale=.001, suffix="K")) + 
  facet_wrap(~rank,
             scales=c("free")) +
  faceted +
  labs(y="",
       title="African-American land ownership in Georgia 1899, by county")
